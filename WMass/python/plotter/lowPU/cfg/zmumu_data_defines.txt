

Muon_pt_corr        : applyRochesterData(Muon_pt, Muon_eta, Muon_phi, Muon_charge)
vetoMuons           : Muon_pt_corr > 10 && Muon_looseId && abs(Muon_eta) < 2.4 && abs(Muon_dxybs) < 0.05 && abs(Muon_dz)< 0.2


vetoElectrons       : Electron_pt > 10 && Electron_cutBased > 0 && abs(Electron_eta) < 2.4 && abs(Electron_dxy) < 0.05 && abs(Electron_dz)< 0.2
goodMuons           : vetoMuons && Muon_pt_corr > 25 && Muon_tightId && Muon_pfRelIso04_all < 0.15 
# && Muon_mediumId
goodTrigObjs        : goodMuonTriggerCandidateLowPU(TrigObj_id,TrigObj_pt,TrigObj_l1pt,TrigObj_l2pt,TrigObj_filterBits)
goodMuonsPlus       : goodMuons && Muon_charge > 0
goodMuonsMinus      : goodMuons && Muon_charge < 0

leadingMuon         : goodMuons[0] ? Muon_pt_corr[goodMuons[0]] >  Muon_pt_corr[goodMuons[1]] : goodMuons[1]
trailingMuon        : goodMuons[1]

m_mumu              : invariantmass(Muon_pt_corr[goodMuons],Muon_eta[goodMuons],Muon_phi[goodMuons],Muon_mass[goodMuons])
pt_mumu             : transversemomentumBoson(Muon_pt_corr[goodMuons],Muon_eta[goodMuons],Muon_phi[goodMuons],Muon_mass[goodMuons])
leadingMuonPt       : Muon_pt_corr[goodMuons][0]
subLeadingMuonPt    : Muon_pt_corr[goodMuons][1]

trigMatch           : hasTriggerMatch(Muon_eta[goodMuons], Muon_phi[goodMuons], TrigObj_eta[goodTrigObjs], TrigObj_phi[goodTrigObjs])
lepSF               : 1
